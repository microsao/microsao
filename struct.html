结构体排序

#include

#define N 3

struct
{
int num;
float math;
float chinese;
float english;
float sum;
}st[N],man;

int main()
{
int i,j,k;
for (i=0;i {
printf ("please input student%d's num math chinese english score\n",i+1);
scanf ("%d%f%f%f",&st[i].num,&st[i].math,&st[i].chinese,&st[i].english);
st[i].sum=st[i].math+st[i].chinese+st[i].english;
}
for (i=0;i {
k=i;
for (j=i+1;j {
if (st[j].sum>st[k].sum)
{
k=j;
man=st[k];
st[k]=st[i];
st[i]=man;
}
else if (st[j].sum=st[k].sum)
{
if (st[j].english>st[k].english)
{
k=j;
man=st[k];
st[k]=st[i];
st[i]=man;

}
}
}
}
printf ("num math chinese english sum\n");
for (i=0;i {
printf ("%-6d%-4.2f%-6.2f%-6.2f%-4.2f\n",st[i].num,st[i].math,st[i].chinese,st[i].english,st[i].sum);
}
return 0;
}
