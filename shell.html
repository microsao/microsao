#!/bin/bash

UPHOST=/root/Desktop/shell/uphost
DOWNHOST=/root/Desktop/shell/downhost

fip="192.168.199"
for ip in `seq 1 254`
do
echo "pinging ${fip}.${ip}..."
ping -c1 -w1 ${fip}.${ip} > /dev/null 2>&1
if [ "$?" -eq "0" ]; then
echo "ok"
echo "${fip}.${ip}" >> ${UPHOST}
else
echo "failed"
echo "${fip}.${ip}" >> ${DOWNHOST}
fi
done
