struct student * del(struct student *head,int x)
{
struct student *p,*q=NULL;
if(x <= 0)
return head;

for(p = head; p && (x != p->num); p=p->next)
q = p;

if(q!= NULL && q->next ==NULL)
{
printf("\n\nhave not this student !!!");
return head;
}
else if(q == NULL)
{
p = head;
head = p->next;
free(p);
}
else
{
p = q->next;
q->next = p->next;
free(p);
}
return head;
}
